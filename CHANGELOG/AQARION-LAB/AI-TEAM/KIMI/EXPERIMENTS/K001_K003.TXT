class KIMIS_K001:
    """
    KIMIS Experiment K-001: Observable Defect Refinement Stress Test
    
    Tests whether defect operator D_Π = (I-P_Π)U_T P_Π behaves as predicted
    under observable refinement.
    
    Main Hypothesis H1: Refinement monotonicity
    For Π₁ ⪯ Π₂ (refinement), expect ||D_Π₂|| ≤ ||D_Π₁||
    """
    
    def __init__(self, n_max: int = 5):
        self.n_max = n_max
        self.systems_tested = 0
        self.partitions_tested = 0
        self.counterexamples = []
        self.results = []
        self.systems = []
        self.partitions = []
        
    def generate_all_maps(self, n: int) -> List[np.ndarray]:
        """Generate all deterministic maps T: {0,...,n-1} -> {0,...,n-1}."""
        # For n=5, this is 5^5 = 3125 maps — manageable
        maps = []
        for values in product(range(n), repeat=n):
            T = np.array(values, dtype=int)
            maps.append(T)
        return maps
    
    def generate_all_partitions(self, n: int) -> List[List[Set[int]]]:
        """Generate all partitions of {0,...,n-1} using Bell numbers."""
        # Use recursive generation
        def partitions_of_set(s: Set[int]) -> List[List[Set[int]]]:
            if not s:
                return [[]]
            elem = min(s)
            rest = s - {elem}
            result = []
            for partition in partitions_of_set(rest):
                # Add elem to existing block
                for i, block in enumerate(partition):
                    new_partition = [b.copy() for b in partition]
                    new_partition[i] = new_partition[i] | {elem}
                    result.append(new_partition)
                # Add elem as new block
                result.append([{elem}] + [b.copy() for b in partition])
            return result
        
        return partitions_of_set(set(range(n)))
    
    def build_koopman(self, T: np.ndarray, n: int) -> np.ndarray:
        """Build Koopman operator U_T as column-stochastic matrix."""
        U = np.zeros((n, n))
        for i in range(n):
            j = T[i]
            U[j, i] = 1.0
        return U
    
    def build_projection(self, partition: List[Set[int]], n: int) -> np.ndarray:
        """Build projection P_Π onto observables constant on blocks."""
        P = np.zeros((n, n))
        for block in partition:
            block_list = list(block)
            k = len(block_list)
            weight = 1.0 / k
            for i in block_list:
                for j in block_list:
                    P[i, j] = weight
        return P
    
    def compute_defect(self, U: np.ndarray, P: np.ndarray) -> np.ndarray:
        """Compute D_Π = (I - P) U P."""
        I = np.eye(U.shape[0])
        return (I - P) @ U @ P
    
    def is_exact_quotient(self, T: np.ndarray, partition: List[Set[int]], n: int) -> bool:
        """Direct test: does T map each block into a single block?"""
        block_map = {}
        for idx, block in enumerate(partition):
            for elem in block:
                block_map[elem] = idx
        
        for block in partition:
            targets = set()
            for elem in block:
                targets.add(block_map[T[elem]])
            if len(targets) > 1:
                return False
        return True
    
    def frobenius_norm(self, D: np.ndarray) -> float:
        return float(np.sqrt(np.sum(D * D)))
    
    def spectral_norm(self, D: np.ndarray) -> float:
        return float(np.linalg.norm(D, ord=2))
    
    def defect_rank(self, D: np.ndarray) -> int:
        return int(np.linalg.matrix_rank(D, tol=1e-10))
    
    def scc_count(self, T: np.ndarray, n: int) -> int:
        """Count strongly connected components using Tarjan."""
        visited = [False] * n
        on_stack = [False] * n
        index = [0] * n
        lowlink = [0] * n
        idx = [0]
        stack = []
        sccs = []
        
        def strongconnect(v: int):
            index[v] = idx[0]
            lowlink[v] = idx[0]
            idx[0] += 1
            stack.append(v)
            on_stack[v] = True
            w = T[v]
            if not visited[w]:
                visited[w] = True
                strongconnect(w)
                lowlink[v] = min(lowlink[v], lowlink[w])
            elif on_stack[w]:
                lowlink[v] = min(lowlink[v], index[w])
            if lowlink[v] == index[v]:
                scc = []
                while True:
                    w = stack.pop()
                    on_stack[w] = False
                    scc.append(w)
                    if w == v:
                        break
                sccs.append(scc)
        
        for v in range(n):
            if not visited[v]:
                visited[v] = True
                strongconnect(v)
        return len(sccs)
    
    def max_transient_depth(self, T: np.ndarray, n: int) -> int:
        """Compute maximum depth of transient states."""
        # Find cycle states
        in_cycle = [False] * n
        visited = [False] * n
        
        for start in range(n):
            if visited[start]:
                continue
            path = []
            current = start
            while True:
                if visited[current]:
                    if current in path:
                        cycle_start = path.index(current)
                        for node in path[cycle_start:]:
                            in_cycle[node] = True
                    break
                visited[current] = True
                path.append(current)
                current = T[current]
        
        # Compute depths
        depth = {}
        def get_depth(v: int, visited_set=None):
            if visited_set is None:
                visited_set = set()
            if v in depth:
                return depth[v]
            if v in visited_set:
                return 0
            visited_set.add(v)
            if in_cycle[v]:
                depth[v] = 0
                return 0
            w = T[v]
            d = 1 + get_depth(w, visited_set)
            depth[v] = d
            return d
        
        for v in range(n):
            get_depth(v)
        
        return max(depth.values()) if depth else 0
    
    def is_refinement(self, P1: List[Set[int]], P2: List[Set[int]]) -> bool:
        """Check if P1 refines P2 (P1 is finer than P2)."""
        # Every block of P1 is contained in some block of P2
        for b1 in P1:
            found = False
            for b2 in P2:
                if b1.issubset(b2):
                    found = True
                    break
            if not found:
                return False
        return True

                    P = self.build_projection(partition, n)

                    defect_zero = np.allclose(D, 0, atol=1e-10)
                    exact = self.is_exact_quotient(T, partition, n)
                    
                    if defect_zero != exact:

class KIMIS_K001:
    """
    KIMIS Experiment K-001: Observable Defect Refinement Stress Test
    
    Tests whether defect operator D_Π = (I-P_Π)U_T P_Π behaves as predicted
    under observable refinement.
    
    Main Hypothesis H1: Refinement monotonicity
    For Π₁ ⪯ Π₂ (refinement), expect ||D_Π₂|| ≤ ||D_Π₁||
    """
    
    def __init__(self, n_max: int = 5):
        self.n_max = n_max
        self.systems_tested = 0
        self.partitions_tested = 0
        self.counterexamples = []
        self.results = []
        self.systems = []
        self.partitions = []
        
    def generate_all_maps(self, n: int) -> List[np.ndarray]:
        """Generate all deterministic maps T: {0,...,n-1} -> {0,...,n-1}."""
        # For n=5, this is 5^5 = 3125 maps — manageable
        maps = []
        for values in product(range(n), repeat=n):
            T = np.array(values, dtype=int)
            maps.append(T)
        return maps
    
    def generate_all_partitions(self, n: int) -> List[List[Set[int]]]:
        """Generate all partitions of {0,...,n-1} using Bell numbers."""
        # Use recursive generation
        def partitions_of_set(s: Set[int]) -> List[List[Set[int]]]:
            if not s:
                return [[]]
            elem = min(s)
            rest = s - {elem}
            result = []
            for partition in partitions_of_set(rest):
                # Add elem to existing block
                for i, block in enumerate(partition):
                    new_partition = [b.copy() for b in partition]
                    new_partition[i] = new_partition[i] | {elem}
                    result.append(new_partition)
                # Add elem as new block
                result.append([{elem}] + [b.copy() for b in partition])
            return result
        
        return partitions_of_set(set(range(n)))
    
    def build_koopman(self, T: np.ndarray, n: int) -> np.ndarray:
        """Build Koopman operator U_T as column-stochastic matrix."""
        U = np.zeros((n, n))
        for i in range(n):
            j = T[i]
            U[j, i] = 1.0
        return U
    
    def build_projection(self, partition: List[Set[int]], n: int) -> np.ndarray:
        """Build projection P_Π onto observables constant on blocks."""
        P = np.zeros((n, n))
        for block in partition:
            block_list = list(block)
            k = len(block_list)
            weight = 1.0 / k
            for i in block_list:
                for j in block_list:
                    P[i, j] = weight
        return P
    
    def compute_defect(self, U: np.ndarray, P: np.ndarray) -> np.ndarray:
        """Compute D_Π = (I - P) U P."""
        I = np.eye(U.shape[0])
        return (I - P) @ U @ P
    
    def is_exact_quotient(self, T: np.ndarray, partition: List[Set[int]], n: int) -> bool:
        """Direct test: does T map each block into a single block?"""
        block_map = {}
        for idx, block in enumerate(partition):
            for elem in block:
                block_map[elem] = idx
        
        for block in partition:
            targets = set()
            for elem in block:
                targets.add(block_map[T[elem]])
            if len(targets) > 1:
                return False
        return True
    
    def frobenius_norm(self, D: np.ndarray) -> float:
        return float(np.sqrt(np.sum(D * D)))
    
    def spectral_norm(self, D: np.ndarray) -> float:
        return float(np.linalg.norm(D, ord=2))
    
    def defect_rank(self, D: np.ndarray) -> int:
        return int(np.linalg.matrix_rank(D, tol=1e-10))
    
    def scc_count(self, T: np.ndarray, n: int) -> int:
        """Count strongly connected components using Tarjan."""
        visited = [False] * n
        on_stack = [False] * n
        index = [0] * n
        lowlink = [0] * n
        idx = [0]
        stack = []
        sccs = []
        
        def strongconnect(v: int):
            index[v] = idx[0]
            lowlink[v] = idx[0]
            idx[0] += 1
            stack.append(v)
            on_stack[v] = True
            w = T[v]
            if not visited[w]:
                visited[w] = True
                strongconnect(w)
                lowlink[v] = min(lowlink[v], lowlink[w])
            elif on_stack[w]:
                lowlink[v] = min(lowlink[v], index[w])
            if lowlink[v] == index[v]:
                scc = []
                while True:
                    w = stack.pop()
                    on_stack[w] = False
                    scc.append(w)
                    if w == v:
                        break
                sccs.append(scc)
        
        for v in range(n):
            if not visited[v]:
                visited[v] = True
                strongconnect(v)
        return len(sccs)
    
    def max_transient_depth(self, T: np.ndarray, n: int) -> int:
        """Compute maximum depth of transient states."""
        # Find cycle states
        in_cycle = [False] * n
        visited = [False] * n
        
        for start in range(n):
            if visited[start]:
                continue
            path = []
            current = start
            while True:
                if visited[current]:
                    if current in path:
                        cycle_start = path.index(current)
                        for node in path[cycle_start:]:
                            in_cycle[node] = True
                    break
                visited[current] = True
                path.append(current)
                current = T[current]
        
        # Compute depths
        depth = {}
        def get_depth(v: int, visited_set=None):
            if visited_set is None:
                visited_set = set()
            if v in depth:
                return depth[v]
            if v in visited_set:
                return 0
            visited_set.add(v)
            if in_cycle[v]:
                depth[v] = 0
                return 0
            w = T[v]
            d = 1 + get_depth(w, visited_set)
            depth[v] = d
            return d
        
        for v in range(n):
            get_depth(v)
        
        return max(depth.values()) if depth else 0
    
    def is_refinement(self, P1: List[Set[int]], P2: List[Set[int]]) -> bool:
        """Check if P1 refines P2 (P1 is finer than P2)."""
        # Every block of P1 is contained in some block of P2
        for b1 in P1:
            found = False
            for b2 in P2:
                if b1.issubset(b2):
                    found = True
                    break
            if not found:
                return False
        return True
    
    def run_experiment(self) -> Dict:
        print("=" * 70)
        print("KIMIS Experiment Run K-001")
        print("Observable Defect Refinement Stress Test")
        print("=" * 70)
        print(f"Date: {datetime.now(timezone.utc).isoformat()}")
        print(f"Max n: {self.n_max}")
        print()
        
        for n in range(2, self.n_max + 1):
            print(f"\n{'─' * 50}")
            print(f"System size: n = {n}")
            print(f"{'─' * 50}")
            
            # Generate all maps and partitions
            maps = self.generate_all_maps(n)
            partitions = self.generate_all_partitions(n)
            
            print(f"  Maps: {len(maps)}")
            print(f"  Partitions: {len(partitions)}")
            
            h1_violations = 0
            h1_tests = 0
            exact_mismatches = 0
            exact_tests = 0
            
            for T in maps:
                U = self.build_koopman(T, n)
                
                # Test 1: Exact quotient detection
                for partition in partitions:
                    P = self.build_projection(partition, n)
                    D = self.compute_defect(U, P)
                    
                    defect_zero = np.allclose(D, 0, atol=1e-10)
                    exact = self.is_exact_quotient(T, partition, n)
                    
                    if defect_zero != exact:
                        exact_mismatches += 1
                    exact_tests += 1
                
                # Test H1: Refinement monotonicity
                # Sample pairs of partitions where one refines the other
                for i, P1 in enumerate(partitions):
                    for j, P2 in enumerate(partitions):
                        if i == j:
                            continue
                        if self.is_refinement(P1, P2):
                            P1_mat = self.build_projection(P1, n)
                            P2_mat = self.build_projection(P2, n)
                            D1 = self.compute_defect(U, P1_mat)
                            D2 = self.compute_defect(U, P2_mat)
                            
                            norm1 = self.frobenius_norm(D1)
                            norm2 = self.frobenius_norm(D2)
                            rank1 = self.defect_rank(D1)
                            rank2 = self.defect_rank(D2)
                            
                            h1_tests += 1
                            
                            # H1: ||D2|| <= ||D1||
                            if norm2 > norm1 + 1e-10:
                                h1_violations += 1
                                self.counterexamples.append({
                                    "n": n,
                                    "T": T.tolist(),
                                    "P1": [list(b) for b in P1],
                                    "P2": [list(b) for b in P2],
                                    "norm1": norm1,
                                    "norm2": norm2,
                                    "rank1": rank1,
                                    "rank2": rank2
                                })
                
                self.systems_tested += 1
            
            self.partitions_tested += len(partitions) * len(maps)
            
            print(f"  Exact mismatch rate: {exact_mismatches}/{exact_tests} ({100*exact_mismatches/max(exact_tests,1):.4f}%)")
            print(f"  H1 violations: {h1_violations}/{h1_tests} ({100*h1_violations/max(h1_tests,1):.4f}%)")
            
            # Store results
            self.results.append({
                "n": n,
                "maps": len(maps),
                "partitions": len(partitions),
                "exact_mismatches": exact_mismatches,
                "exact_tests": exact_tests,
                "h1_violations": h1_violations,
                "h1_tests": h1_tests
            })
        
        # Build certificate
        return self.generate_certificate()
    
    def generate_certificate(self) -> Dict:
        """Generate the KIMIS-K001 certificate."""
        payload = json.dumps({
            "results": self.results,
            "counterexamples": self.counterexamples
        }, sort_keys=True)
        
        certificate = {
            "experiment": "KIMIS-K001",
            "hypothesis": "refinement_monotonicity",
            "version": "0.4.1-final",
            "release_hash": "d8152b3cca0dd91e5bd0b519589cb79c5574ef079964a9e822267206128f7230",

            "systems_tested": self.systems_tested,
            "partitions_tested": self.partitions_tested,
            "counterexamples": len(self.counterexamples),
            "status": "REFUTED" if self.counterexamples else "VERIFIED",
            "sha256": hashlib.sha256(payload.encode()).hexdigest()
        }
        
        return certificate

def generate_partitions(n):
"""Generate all set partitions of {0, ..., n-1}."""
if n == 0:
yield []
return

# Use restricted growth strings  
def helper(current, max_label):  
    if len(current) == n:  
        # Convert RGS to partition  
        partition = [[] for _ in range(max_label + 1)]  
        for i, label in enumerate(current):  
            partition[label].append(i)  
        yield [tuple(sorted(block)) for block in partition if block]  
        return  
      
    for label in range(max_label + 2):  
        if label <= max_label + 1:  
            new_max = max(max_label, label)  
            yield from helper(current + [label], new_max)  
  
yield from helper([], -1)

def meet_partition(p1, p2):
"""Meet (finest common coarsening) of two partitions."""
# Two elements are in the same block of meet if they are in the same
# block of both p1 and p2
n = max(max(block) for block in p1) + 1

# Union-find  
parent = list(range(n))  
  
def find(x):  
    if parent[x] != x:  
        parent[x] = find(parent[x])  
    return parent[x]  
  
def union(x, y):  
    px, py = find(x), find(y)  
    if px != py:  
        parent[px] = py  
  
# Union elements in same block of p1  
for block in p1:  
    for i in range(1, len(block)):  
        union(block[0], block[i])  
  
# Union elements in same block of p2  
for block in p2:  
    for i in range(1, len(block)):  
        union(block[0], block[i])  
  
# Build partition from union-find  
blocks = {}  
for i in range(n):  
    root = find(i)  
    if root not in blocks:  
        blocks[root] = []  
    blocks[root].append(i)  
  
return tuple(sorted(tuple(sorted(block)) for block in blocks.values()))

def join_partition(p1, p2):
"""Join (coarsest common refinement) of two partitions."""
# Two elements are in the same block of join if there's a chain
# connecting them through blocks of p1 and p2
n = max(max(block) for block in p1) + 1

# Build adjacency: connect elements in same block of p1 or p2  
from collections import defaultdict  
adj = defaultdict(set)  
  
for block in p1:  
    for i in block:  
        for j in block:  
            adj[i].add(j)  
  
for block in p2:  
    for i in block:  
        for j in block:  
            adj[i].add(j)  
  
# Find connected components  
visited = [False] * n  
components = []  
  
for start in range(n):  
    if not visited[start]:  
        component = []  
        stack = [start]  
        while stack:  
            node = stack.pop()  
            if not visited[node]:  
                visited[node] = True  
                component.append(node)  
                for neighbor in adj[node]:  
                    if not visited[neighbor]:  
                        stack.append(neighbor)  
        components.append(tuple(sorted(component)))  
  
return tuple(sorted(components))

KIMIS-K002: Exact Observable Lattice Experiment

Characterizes the algebraic structure of:
E(T) = {Π | D_Π = 0}

Primary Questions:
Q1: Do exact observables form a lattice?
Q2: Find the canonical minimal observable quotient
Q3: Measure observable landscape
"""

def init(self, n_max: int = 5):
self.n_max = n_max
self.results = []
self.counterexamples = []

def generate_all_maps(self, n: int) -> List[np.ndarray]:
maps = []
for values in product(range(n), repeat=n):
T = np.array(values, dtype=int)
maps.append(T)
return maps

def generate_all_partitions(self, n: int) -> List[List[Set[int]]]:
def partitions_of_set(s: Set[int]) -> List[List[Set[int]]]:
if not s:
return [[]]
elem = min(s)
rest = s - {elem}
result = []
for partition in partitions_of_set(rest):
for i, block in enumerate(partition):
new_partition = [b.copy() for b in partition]
new_partition[i] = new_partition[i] | {elem}
result.append(new_partition)
result.append([{elem}] + [b.copy() for b in partition])
return result
return partitions_of_set(set(range(n)))

def build_koopman(self, T: np.ndarray, n: int) -> np.ndarray:
U = np.zeros((n, n))
for i in range(n):
j = T[i]
U[i, j] = 1.0
return U

def build_projection(self, partition: List[Set[int]], n: int) -> np.ndarray:
P = np.zeros((n, n))
for block in partition:
block_list = list(block)
k = len(block_list)
weight = 1.0 / k
for i in block_list:
for j in block_list:
P[i, j] = weight
return P

def compute_defect(self, U: np.ndarray, P: np.ndarray) -> np.ndarray:
I = np.eye(U.shape[0])
return (I - P) @ U @ P

def is_exact(self, T: np.ndarray, partition: List[Set[int]], n: int) -> bool:
"""Direct test: does T map each block into a single block?"""
block_map = {}
for idx, block in enumerate(partition):
for elem in block:
block_map[elem] = idx

for block in partition:    
    targets = set()    
    for elem in block:    
        targets.add(block_map[int(T[elem])])    
    if len(targets) > 1:    
        return False    
return True

def partition_meet(self, P1: List[Set[int]], P2: List[Set[int]]) -> List[Set[int]]:
"""Compute meet (coarsest common refinement) of two partitions."""
# Meet = coarsest partition that refines both
# = partition into intersections of blocks
result = []
used = set()

for b1 in P1:    
    for b2 in P2:    
        inter = b1 & b2    
        if inter and inter not in used:    
            result.append(inter)    
            used.add(frozenset(inter))    
    
# Remove duplicates and empty sets    
result = [b for b in result if b]    
return result

def partition_join(self, P1: List[Set[int]], P2: List[Set[int]], n: int) -> List[Set[int]]:
"""Compute join (finest common coarsening) of two partitions."""
# Join = finest partition that is coarser than both
# = connected components of the union of equivalence relations
# Build equivalence relation: x ~ y if they are in the same block in P1 or P2
equiv = {i: {i} for i in range(n)}

def find(x):    
    if equiv[x] != {x}:    
        # Find representative    
        rep = min(equiv[x])    
        return rep    
    return x    
    
def union(x, y):    
    rx, ry = find(x), find(y)    
    if rx != ry:    
        equiv[rx] = equiv[rx] | equiv[ry]    
        for z in equiv[ry]:    
            equiv[z] = equiv[rx]    
    
for partition in [P1, P2]:    
    for block in partition:    
        block_list = sorted(list(block))    
        for i in range(len(block_list) - 1):    
            union(block_list[i], block_list[i+1])    
    
# Extract blocks    
blocks = []    
seen = set()    
for i in range(n):    
    rep = find(i)    
    if rep not in seen:    
        seen.add(rep)    
        blocks.append(equiv[rep])    
    
return [b for b in blocks if b]

def is_refinement(self, P1: List[Set[int]], P2: List[Set[int]]) -> bool:
"""Check if P1 refines P2 (P1 is finer than P2)."""
for b1 in P1:
found = False
for b2 in P2:
if b1.issubset(b2):
found = True
break
if not found:
return False
return True

class KIMIS_K003:
    """
    KIMIS-K003: Exact Observable Spectrum + Lattice Classification
    
    Classifies the internal geometry of the exact observable lattice E(T).
    
    Key insight: For deterministic systems, the identity partition is always exact,
    so "minimal quotient" is trivial. The interesting object is the SPECTRUM of
    exact observable scales and the lattice geometry.
    """
    
    def __init__(self, n_max: int = 5):
        self.n_max = n_max
        self.results = []
        
    def generate_all_maps(self, n: int) -> List[np.ndarray]:
        maps = []
        for values in product(range(n), repeat=n):
            T = np.array(values, dtype=int)
            maps.append(T)
        return maps
    
    def generate_all_partitions(self, n: int) -> List[List[Set[int]]]:
        def partitions_of_set(s: Set[int]) -> List[List[Set[int]]]:
            if not s:
                return [[]]
            elem = min(s)
            rest = s - {elem}
            result = []
            for partition in partitions_of_set(rest):
                for i, block in enumerate(partition):
                    new_partition = [b.copy() for b in partition]
                    new_partition[i] = new_partition[i] | {elem}
                    result.append(new_partition)
                result.append([{elem}] + [b.copy() for b in partition])
            return result
        return partitions_of_set(set(range(n)))
    
    def is_exact(self, T: np.ndarray, partition: List[Set[int]], n: int) -> bool:
        """Test if partition is exact: T maps each block into a single block."""
        block_map = {}
        for idx, block in enumerate(partition):
            for elem in block:
                block_map[elem] = idx
        
        for block in partition:
            targets = set()
            for elem in block:
                targets.add(block_map[int(T[elem])])
            if len(targets) > 1:
                return False
        return True
    
    def partition_meet(self, P1: List[Set[int]], P2: List[Set[int]]) -> List[Set[int]]:
        """Compute meet (coarsest common refinement): intersections of blocks."""
        result = []
        used = set()
        
        for b1 in P1:
            for b2 in P2:
                inter = b1 & b2
                if inter and frozenset(inter) not in used:
                    result.append(inter)
                    used.add(frozenset(inter))
        
        return [b for b in result if b]
    
    def partition_join(self, P1: List[Set[int]], P2: List[Set[int]], n: int) -> List[Set[int]]:
        """Compute join (finest common coarsening): connected components of union."""
        # Build union-find
        parent = {i: i for i in range(n)}
        
        def find(x):
            if parent[x] != x:
                parent[x] = find(parent[x])
            return parent[x]
        
        def union(x, y):
            rx, ry = find(x), find(y)
            if rx != ry:
                parent[rx] = ry
        
        # Union elements in same block of either partition
        for partition in [P1, P2]:
            for block in partition:
                block_list = sorted(list(block))
                for i in range(len(block_list) - 1):
                    union(block_list[i], block_list[i + 1])
        
        # Extract blocks
        blocks = defaultdict(set)
        for i in range(n):
            blocks[find(i)].add(i)
        
        return [b for b in blocks.values() if b]
    
    def is_refinement(self, P1: List[Set[int]], P2: List[Set[int]]) -> bool:
        """Check if P1 refines P2 (P1 is finer than P2)."""
        for b1 in P1:
            found = False
            for b2 in P2:
                if b1.issubset(b2):
                    found = True
                    break
            if not found:
                return False
        return True
    
    def compute_lattice_invariants(self, exact_partitions: List[List[Set[int]]], n: int) -> Dict:
        """Compute height, width, atoms, coatoms of the exact observable lattice."""
        if not exact_partitions:
            return {"height": 0, "width": 0, "atoms": 0, "coatoms": 0}
        
        # Height: longest refinement chain
        # Build DAG of refinement relations
        # Sort by number of blocks (finer = more blocks)
        sorted_partitions = sorted(exact_partitions, key=lambda p: len(p))
        
        # Dynamic programming for longest chain ending at each partition
        longest_chain = {}
        for i, P in enumerate(sorted_partitions):
            longest_chain[i] = 1
            for j in range(i):
                Q = sorted_partitions[j]
                if self.is_refinement(Q, P):  # Q refines P (Q is finer)
                    longest_chain[i] = max(longest_chain[i], longest_chain[j] + 1)
        
        height = max(longest_chain.values()) if longest_chain else 0
        
        # Width: largest antichain (partitions where none refines another)
        # Use brute force for small n
        max_antichain = 0
        for r in range(1, len(exact_partitions) + 1):
            for subset in combinations(range(len(exact_partitions)), r):
                is_antichain = True
                for i, j in combinations(subset, 2):
                    Pi = exact_partitions[i]
                    Pj = exact_partitions[j]
                    if self.is_refinement(Pi, Pj) or self.is_refinement(Pj, Pi):
                        is_antichain = False
                        break
                if is_antichain:
                    max_antichain = max(max_antichain, r)
        
        # Atoms: partitions immediately above the bottom (coarsest partition)
        # In refinement order, bottom is coarsest (fewest blocks)
        # Atoms are partitions that cover the bottom
        coarsest = [set(range(n))]
        atoms = []
        for P in exact_partitions:
            if len(P) == 2:  # Immediately above coarsest (1 block -> 2 blocks)
                # Verify P is exact and covers coarsest
                if self.is_refinement(P, coarsest) and len(P) == 2:
                    atoms.append(P)
        
        # Coatoms: partitions immediately below the top (identity partition)
        identity = [{i} for i in range(n)]
        coatoms = []
        for P in exact_partitions:
            if len(P) == n - 1:  # Immediately below identity
                if self.is_refinement(identity, P) and len(P) == n - 1:
                    coatoms.append(P)
        
        return {
            "height": height,
            "width": max_antichain,
            "atoms": len(atoms),
            "coatoms": len(coatoms),
            "exact_count": len(exact_partitions)
        }
    
    def test_distributivity(self, exact_partitions: List[List[Set[int]]], T: np.ndarray, n: int) -> Dict:
        """Test if the exact observable lattice is distributive."""
        if len(exact_partitions) < 3:
            return {"distributive": True, "tests": 0, "violations": 0}
        
        tests = 0
        violations = 0
        
        # Test a ∧ (b ∨ c) = (a ∧ b) ∨ (a ∧ c)
        for a_idx in range(min(len(exact_partitions), 10)):  # Sample for efficiency
            for b_idx in range(min(len(exact_partitions), 10)):
                for c_idx in range(min(len(exact_partitions), 10)):
                    a = exact_partitions[a_idx]
                    b = exact_partitions[b_idx]
                    c = exact_partitions[c_idx]
                    
                    # Compute both sides
                    left_meet = self.partition_meet(a, self.partition_join(b, c, n))
                    right_join = self.partition_join(self.partition_meet(a, b), self.partition_meet(a, c), n)
                    
                    # Check if both are exact and equal
                    left_exact = self.is_exact(T, left_meet, n)
                    right_exact = self.is_exact(T, right_join, n)
                    
                    if left_exact and right_exact:
                        tests += 1
                        # Compare as sorted tuples
                        left_sorted = sorted([sorted(list(b)) for b in left_meet])
                        right_sorted = sorted([sorted(list(b)) for b in right_join])
                        if left_sorted != right_sorted:
                            violations += 1
                    
                    # Also test dual: a ∨ (b ∧ c) = (a ∨ b) ∧ (a ∨ c)
                    left_join = self.partition_join(a, self.partition_meet(b, c), n)
                    right_meet = self.partition_meet(self.partition_join(a, b, n), self.partition_join(a, c, n))
                    
                    left_exact2 = self.is_exact(T, left_join, n)
                    right_exact2 = self.is_exact(T, right_meet, n)
                    
                    if left_exact2 and right_exact2:
                        tests += 1
                        left_sorted2 = sorted([sorted(list(b)) for b in left_join])
                        right_sorted2 = sorted([sorted(list(b)) for b in right_meet])
                        if left_sorted2 != right_sorted2:
                            violations += 1
        
        return {
            "distributive": violations == 0,
            "tests": tests,
            "violations": violations
        }
    
    def test_modularity(self, exact_partitions: List[List[Set[int]]], T: np.ndarray, n: int) -> Dict:
        """Test if the exact observable lattice is modular."""
        if len(exact_partitions) < 3:
            return {"modular": True, "tests": 0, "violations": 0}
        
        tests = 0
        violations = 0
        
        # Test: if a ≤ c, then a ∨ (b ∧ c) = (a ∨ b) ∧ c
        for a_idx in range(min(len(exact_partitions), 10)):
            for c_idx in range(min(len(exact_partitions), 10)):
                a = exact_partitions[a_idx]
                c = exact_partitions[c_idx]
                
                # Check if a ≤ c (a refines c)
                if not self.is_refinement(a, c):
                    continue
                
                for b_idx in range(min(len(exact_partitions), 10)):
                    b = exact_partitions[b_idx]
                    
                    left = self.partition_join(a, self.partition_meet(b, c), n)
                    right = self.partition_meet(self.partition_join(a, b, n), c)
                    
                    left_exact = self.is_exact(T, left, n)
                    right_exact = self.is_exact(T, right, n)
                    
                    if left_exact and right_exact:
                        tests += 1
                        left_sorted = sorted([sorted(list(b)) for b in left])
                        right_sorted = sorted([sorted(list(b)) for b in right])
                        if left_sorted != right_sorted:
                            violations += 1
        
        return {
            "modular": violations == 0,
            "tests": tests,
            "violations": violations
        }
